#### 1.1 Dynamic Tool Page Metadata
**File:** `app/tool/[slug]/page.tsx`

**Requirements:**
- Auto-generate SEO Title: `{Tool Name} Online Free – Fast & Secure {Keyword} Tool`
- Auto-generate Meta Description: `Use our free {tool} tool to instantly convert, compress or generate. No login required. Fast & secure.`
- URL Structure: `/image-to-jpg-converter`, `/csv-to-excel-converter`, `/free-url-shortener`
- Add FAQ schema for rich results
- Add JSON-LD structured data

**Implementation:**
```typescript
export async function generateMetadata({ params }): Promise<Metadata> {
  const { slug } = await params;
  const tool = await getToolBySlug(slug);
  
  const seoTitle = `${tool.name} Online Free – Fast & Secure ${tool.keyword} Tool`;
  const seoDesc = `Use our free ${tool.name.toLowerCase()} tool to instantly convert, compress or generate. No login required. Fast & secure.`;
  
  return {
    title: seoTitle,
    description: seoDesc,
    keywords: tool.seo?.keywords || [],
    openGraph: {
      title: seoTitle,
      description: seoDesc,
      images: [`/og/${tool.slug}.png`],
      url: `https://indiatoolkit.in${tool.slug}`,
    },
    alternates: {
      canonical: `https://indiatoolkit.in${tool.slug}`,
    },
  };
}
```

export default function sitemap() {
  const tools = getAllTools();
  const baseUrl = 'https://indiatoolkit.in';
  
  const toolUrls = tools.map((tool) => ({
    url: `${baseUrl}/tool/${tool.slug}`,
    lastModified: tool.updatedAt || new Date(),
    changeFrequency: 'weekly' as const,
    priority: 0.8
  }));
  
  const categoryUrls = categories.map((cat) => ({
    url: `${baseUrl}/category/${cat.slug}`,
    lastModified: new Date(),
    changeFrequency: 'daily' as const,
    priority: 0.6
  }));
  
  return [
    { url: baseUrl, lastModified: new Date(), priority: 1.0 },
    { url: `${baseUrl}/tools`, lastModified: new Date(), priority: 0.9 },
    ...toolUrls,
    ...categoryUrls
  ];
}
